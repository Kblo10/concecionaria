import mysql.connector

def conectar_banco():
    conn = mysql.connector.connect(
        host="localhost",
        user="root",
        password="123456",
        database="agregacao"
    )
    return conn

# Teste a conexão
conn = conectar_banco()

if conn.is_connected():
    print("Conexão bem-sucedida!")
    conn.close()
else:
    print("Falha na conexão.")
